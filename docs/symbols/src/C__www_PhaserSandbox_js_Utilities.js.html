  2 /**
  3  * Debugging Utilities, has no relation to ingame content!
  4  * @class
  5  * @param {object} public - reference to self for Module Pattern
  6  * @return {object} public - reference to self
  7  */
  8 var Utilities = (function (public) {
  9     
 10     //private vars
 11     var log = true;
 12     var info = true;
 13     var debug = true;
 14     var warn = true;
 15     var error = true;
 16     var showfps = false;
 17     
 18     var expectedGlobals=1000;
 19     
 20     /**
 21     * debug logging wrapper
 22     * @name output
 23     * @memberof Utilities
 24     * @ param {string} message - the text message to use
 25     * @ param {string} type - the type of logging to use (warn,log,error,debug,info)
 26     */
 27     public.output = function (message,type) {
 28         console.warn(message);
 29     };
 30     
 31     /**
 32     * sets up prerequisites for performance monitoring
 33     * @name monitorPerformance
 34     * @memberof Utilities
 35     * @ param {object} currentState - reference to the current game / game state
 36     */
 37     public.monitorPerformance = function (currentState){
 38         showfps=true;
 39         currentState.time.advancedTiming = true;
 40     };
 41     
 42    /**
 43     * checks global variables, excludes common globals, warns if over limit
 44     * @name monitorGlobals
 45     * @memberof Utilities
 46     * @ param {integer} expected - sets the expected number of globals
 47     */
 48     public.monitorGlobals = function (expected){
 49         expectedGlobals=expected || 1000;
 50         var globalsList=Object.keys(window);
 51         for(var i = globalsList.length - 1; i >= 0; i--) {
 52             //add ignore cases here
 53             switch (globalsList[i]){
 54                 case "Utilities":
 55                 case "top":
 56                 case "document":
 57                 case "location":
 58                 case "window":
 59                 case "external":
 60                 case "chrome":
 61                 case "PIXI":
 62                 case "p2":      
 63                 case "Phaser":
 64                     globalsList.splice(i,1);
 65                     break;
 66             }
 67         }
 68         
 69         console.info("Filtered Globals",globalsList);
 70         if ( globalsList.length > expectedGlobals){
 71             console.warn("UTILS: Expecting ", expectedGlobals, " but found ", globalsList.length, " globals");
 72         }
 73             
 74     };
 75 
 76     /**
 77     * Finds locations of tiles, useful for placing objects at coordinates
 78     * based on a tile location, emitters, enemy spawn etc
 79     * @name getTileLocations
 80     * @memberof Utilities
 81     * @ param {object} layer - reference to tilemap layer
 82     * @ param {integer} id - tile id to match
 83     * @ return  array of info for matching tile id from layer 
 84     */
 85     public.getTileLocations=function (layer, id){
 86         var h=0; var v=0; var results = [];
 87         while (v < layer.height){
 88             h=0;
 89             while (h < layer.width){
 90                 if ( layer.data[v][h].index==id ){
 91                     var info={x:layer.data[v][h].worldX ,
 92                               y:layer.data[v][h].worldY,
 93                               width:layer.data[v][h].width,
 94                               height: layer.data[v][h].height,
 95                               tilex:h,
 96                               tiley:v,
 97                               adjustedx:layer.data[v][h].worldX+layer.data[v][h].centerX,
 98                               adjustedy:layer.data[v][h].worldY+layer.data[v][h].centerY
 99                              };
100                     results.push(info)
101                 }
102             h++;
103             }
104         v++;
105         }
106        return results;
107     };
108     
109     return public;
110 
111 }(Utilities || {}));
