  2 /**
  3  * LEVEL1 content manager
  4  * @class
  5  * @param {object} public - reference to self for Module Pattern
  6  * @return {object} public - reference to self
  7  */
  8 THEGAME.LEVEL1 = function (level) {
  9    
 10     var stars , fx;
 11     var scoretext, fpstext;
 12     
 13     /**
 14     * calls base preloader then continues to preload assets
 15     * @name preload
 16     * @memberof THEGAME.LEVEL1
 17     */
 18     level.preload = function () {
 19         THEGAME.preload();
 20         console.info("executing a current level preload");
 21         this.game.load.image('backdrop', 'assets/levels/level1_backdrop.png');
 22         this.game.load.image('layer2', 'assets/levels/level1_pathway.png');
 23         this.game.load.image('foreground', 'assets/levels/level1_foreground.png');
 24         this.game.load.audio('sfx', 'assets/audio/fx_mixdown.ogg');
 25         this.game.load.audio('bg1', 'assets/audio/rain.mp3');
 26         this.game.load.audio('bg2', 'assets/audio/crickets.mp3');
 27         this.game.load.image('wizball', 'assets/sprites/wizball.png');
 28         this.game.load.tilemap('map', 'assets/maps/level1.json', null, Phaser.Tilemap.TILED_JSON);
 29         this.game.load.image('tiles1', 'assets/tiles/tiles1.png');   
 30     };
 31     
 32     /**
 33     * calls base create method then continues to setup stage for level
 34     * @name create
 35     * @memberof THEGAME.LEVEL1
 36     */
 37     level.create = function () { 
 38         
 39         //todo: research if it's necessary to var these variables, 
 40         // they're not appearing globally
 41         THEGAME.create();
 42         console.info("executing a current level create");
 43         
 44         var map = this.game.add.tilemap('map');
 45         map.addTilesetImage('tiles1');
 46         var layer = map.createLayer('tiles');
 47         map.setCollision(1-5);
 48         this.game.physics.p2.convertTilemap(map, layer);
 49         var a = this.game.physics.p2.convertCollisionObjects(map,"collisions");
 50         layer.resizeWorld();
 51         
 52         //setup background layers for parallax
 53         THEGAME.bgsprite = this.game.add.tileSprite(0, 0, this.game.world.width, 768, 'backdrop');
 54         this.game.add.sprite(0, 0, 'layer2');
 55    
 56         THEGAME.setupPlayer(this);
 57         
 58         //setup collectables
 59         stars = this.game.add.group();
 60         stars.enableBody = true;
 61         stars.physicsBodyType = Phaser.Physics.P2JS;
 62         Utilities.getTileLocations (map.layers[0],7).forEach( function (starlocation){
 63             var star = stars.create(starlocation.adjustedx, starlocation.adjustedy, 'star');
 64         });
 65         
 66         var ball = this.game.add.sprite(1000, this.game.world.height - 1000, 'wizball');
 67         this.game.physics.p2.enable(ball);
 68         ball.enableBody=true;
 69         ball.physicsBodyType = Phaser.Physics.P2JS;
 70         ball.body.collideWorldBounds = true;
 71         ball.body.setCircle(30);
 72         ball.scale.setTo(0.7, 0.7)
 73         
 74         var triggers = this.game.add.group();
 75         triggers.enableBody = true;
 76         triggers.physicsBodyType = Phaser.Physics.P2JS;
 77 
 78         var triggerlocations=Utilities.getTileLocations (map.layers[0],5);
 79         triggerlocations.forEach( function (trigger){
 80             var newTrigger = triggers.create(trigger.adjustedx, trigger.adjustedy, 'diamond');
 81             newTrigger.body.static =true;
 82             newTrigger.body.data.shapes[0].sensor=true;
 83         });
 84         
 85         scoretext = this.game.add.text(16, 16, 'score: 0', { fontSize: '32px', fill: '#FFF' });
 86         scoretext.fixedToCamera=true;
 87         fpstext = this.game.add.bitmapText(16, 48, 'speech','fps: 0',32 );
 88         fpstext.fixedToCamera=true;
 89         startext = this.game.add.text(16, 80, 'stars: 0', { fontSize: '32px', fill: '#FFF' });
 90         startext.fixedToCamera=true;
 91         
 92         //audio
 93         var bgsounds1=this.game.add.audio('bg1');
 94         var bgsounds2=this.game.add.audio('bg2');
 95         bgsounds1.volume=0.5;
 96         bgsounds2.volume=0.5;
 97         bgsounds2.play(null, 0, 0.5, true);
 98         bgsounds1.play(null, 0, 0.5, true);
 99         
100         fx = this.game.add.audio('sfx');
101         fx.addMarker('ping', 10, 1.0);
102       
103         
104         this.game.add.sprite(0, 0, 'foreground');
105     };
106     
107     /**
108     * calls base update method then continues level update loop
109     * @name create
110     * @memberof THEGAME.LEVEL1
111     */
112     level.update = function () { 
113         THEGAME.update();
114         
115         fpstext.text = 'fps: '+ this.game.time.fps;
116         startext.text ="stars : "+stars.children.length;
117     };
118     
119     
120     /**
121     * handles player responses to hitting objects (not always response collisions, eg collectables)
122     * @name playerHit
123     * @memberof THEGAME.LEVEL1
124     * @param {object} body - physics body of colliding sprite
125     * @param {object} ShapeA 
126     * @param {object} ShapeB 
127     * @param {object} Equation 
128     */
129     level.playerHit= function (body, shapeA, shapeB, equation) {
130         //not all physics bodys are sprites, check which we are dealing with
131         if (body){
132             if (body.sprite){
133                 switch(body.sprite.key) {
134                     case "star":
135                             this.removeCollectable(body.sprite);
136                         break;
137                     case "diamond":
138                             //do trigger action
139                             this.removeCollectable(body.sprite);
140 
141                         break;
142                      case "wizball":
143                             THEGAME.speak(player, "this is awkward!",20,-20, 5000);
144                         break;
145 
146                     default:
147                         utility.output("no player collision handler in playerHit for "+body.sprite.key);
148                 }
149             } 
150         }
151     };
152     
153     
154     level.removeCollectable= function (star) {
155         //try to generalize this to remove from parent colleciton
156         star.kill();
157         stars.remove(star);     //having trouble removing from array
158         fx.play('ping');
159     };
160         
161   
162     
163     return level;
164 }(THEGAME.LEVEL1 || {});
