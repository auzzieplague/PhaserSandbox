  8 /**
  9  * Includes all common level data, such as player animations and controls

 12 var THEGAME = (function (common) {
 13  
 14     /**
 15     * player object, houses phaser player object
 16     * @memberof THEGAME
 17     * @name player
 18     */
 19     common.player = {};

 23     /**
 24     * preload common assets such as player animations and gui

 26     * @name preload

 41     /**
 42     * setup common elements of game stage such as world physics, player controls
 43     * @memberof THEGAME
 44     * @name create
 45     */
 46     common.create = function () {
 47         console.info("common create method");
 48         
 49         var cs= common.currentstate.game;
 50         cs.stage.disableVisibilityChange = true;
 51         cs.stage.backgroundColor = '#124184';
 52         //cs.world.setBounds(0, 0, 1600, 1200);
 53         cs.physics.startSystem(Phaser.Physics.P2JS);
 54         cs.physics.p2.restitution = 0.0; //bounciness
 55         cs.physics.p2.gravity.y = 1000;
 56         cs.physics.p2.friction = 5;
 57         var worldMaterial = cs.physics.p2.createMaterial('worldMaterial');
 58         //  4 trues = the 4 faces of the world in left, right, top, bottom order
 59         cs.physics.p2.setWorldMaterial(worldMaterial, true, true, true, true);
 60         
 61         //THEGAME.phaser.scale.fullScreenScaleMode = Phaser.ScaleManager.EXACT_FIT;
 62         //THEGAME.phaser.scale.startFullScreen(false);
 63     };
 64     
 65     /**
 66     * setup player sprites, animations and physics body, game camera mode
 67     * @memberof THEGAME
 68     * @name setupPlayer
 69     */
 70     common.setupPlayer = function (currentState) {
 71         
 72         common.player = currentState.game.add.sprite(THEGAME.playerSpawnX,THEGAME.playerSpawnY,'cat');
 73         currentState.game.physics.p2.enable( common.player);
 74         common.player.body.collideWorldBounds = true;
 75         common.player.body.fixedRotation = true;
 76         common.player.anchor.setTo(0.5, 0.5);
 77         common.player.animations.add('walk',  currentState.math.numberArray(0,15), 30, true);
 78         common.player.animations.add('stand',currentState.math.numberArray(20,34), 10, true);
 79         common.player.animations.add('leap',currentState.math.numberArray(42,51 ), 20, false);
 80         common.player.body.onBeginContact.add(currentState.playerHit, currentState);
 81         currentState.game.camera.follow(common.player);
 82         common.player.jumpstate=0;
 83     };    
 84     
 85     
 86     /**
 87     * common update method, includes player controls and general non level specific updates
 88     * @memberof THEGAME
 89     * @name update
 90     */
 91     common.update = function () {
 92         currentState=common.currentstate;
 93         player=common.player;
 94         cursors = currentState.game.input.keyboard.createCursorKeys();
 95             //var player=currentPlayer;
 96             //var player=currentState.player;
 97             
 98             //if player is not jumping
 99             if (player.jumpstate==0 || player.airmove==1){
100                 
101                 //  Reset the players velocity (movement)
102                 //player.body.velocity.x = 0;
103                 
104                 if (cursors.left.isDown || currentState.game.input.keyboard.isDown(Phaser.Keyboard.A)){
105            
106                     player.body.velocity.x = -150;
107                     player.animations.play('walk');
108                     player.scale.setTo(-1,1);
109                 }
110                 else if (cursors.right.isDown || currentState.game.input.keyboard.isDown(Phaser.Keyboard.D)){
111                     //  Move to the right
112                     player.body.velocity.x = 150;
113                     player.animations.play('walk');
114                     player.scale.setTo(1,1); 
115                 }
116                 else{
117                     player.animations.play('stand');
118    
119                 }
120                 
121                 //  Allow the player to jump if they are touching the ground.
122                 if ((   cursors.up.isDown || 
123                         currentState.game.input.keyboard.isDown(Phaser.Keyboard.SPACEBAR)) 
124                         &&  player.jumpstate==0){
125                     
126                     if (player.jumpstate==0){
127                         player.animations.play('leap');
128                         player.jumpstate=1;
129                         player.body.velocity.y = -550;
130                     }
131                 }
132             };
133 
134             if (common.checkIfCanJump()) { 
135                 player.jumpstate=0;
136             };
137         
138             common.updateParallax();
139     };
140     
141     
142     /**
143     * Check to see if player is standing on something, needs more work
144     * can currently jump off of non collidable items such as collectables
145     * @memberof THEGAME
146     * @name checkIfCanJump
147     */
148     common.checkIfCanJump= function () {
149         var yAxis = p2.vec2.fromValues(0, 1);
150         var result = false;
151 
152         for (var i = 0; i < common.currentstate.game.physics.p2.world.narrowphase.contactEquations.length; i++)
153         {
154             var c = common.currentstate.game.physics.p2.world.narrowphase.contactEquations[i];
155 
156             if (c.bodyA === common.player.body.data || c.bodyB === common.player.body.data)
157             {
158                 var d = p2.vec2.dot(c.normalA, yAxis); // Normal dot Y-axis
159                 if (c.bodyA === player.body.data) d *= -1;
160                 if (d > 0.5) result = true;
161             }
162         }
163         return result;
164     }
165     
166     /**
167     * change the offset of the background image based on players current position
168     * @memberof THEGAME
169     * @name updateParallax
170     */
171     common.updateParallax= function (){
172         //could build support for an array of bg images with mutiple scaling
173         //if (THEGAME.player.x>(THEGAME.width/2))
174         THEGAME.bgsprite.tilePosition.x =  THEGAME.player.x*0.2;
175         
176     };
177         
178     return common;
179 }(THEGAME || {}));

184 THEGAME.SpeechBubble = function(x, y, width, text,currentState) {
185         Phaser.Sprite.call(this, currentState, x, y);
186         width = width || 27;
187         var height = 18;
188 
189         // Set up our text and run our custom wrapping routine on it
190         this.bitmapText = currentState.add.bitmapText(x + 12, y + 4, 'speech', text, 22);
191         SpeechBubble.wrapBitmapText(this.bitmapText, width);
192         var bounds = this.bitmapText.getLocalBounds();
193         
194         if (bounds.width + 18 > width) width = bounds.width + 18;
195         if (bounds.height + 14 > height) height = bounds.height + 14;
196         
197         this.borders = [
198             currentState.make.tileSprite(x + 9, y + 9, width - 9, height - 9, 'bubble-border', 4),
199             currentState.make.image(x, y, 'bubble-border', 0),
200             currentState.make.image(x + width, y, 'bubble-border', 2),
201             currentState.make.image(x + width, y + height, 'bubble-border', 8),
202             currentState.make.image(x, y + height, 'bubble-border', 6),
203             currentState.make.tileSprite(x + 9, y, width - 9, 9, 'bubble-border', 1),
204             currentState.make.tileSprite(x + 9, y + height, width - 9, 9, 'bubble-border', 7),
205             currentState.make.tileSprite(x, y + 9, 9, height - 9, 'bubble-border', 3),
206             currentState.make.tileSprite(x + width, y + 9, 9, height - 9, 'bubble-border', 5)
207         ];  
208 
209         for (var b = 0, len = this.borders.length; b < len; b++) {
210             this.addChild(this.borders[b]);   
211         }
212         this.tail = this.addChild(currentState.make.image(x + 18, y + 3 + height, 'bubble-tail'));
213         this.addChild(this.bitmapText);
214         this.bitmapText.tint = 0x111111;
215         this.pivot.set(x + 25, y + height + 24);
216     };
217 
218 THEGAME.SpeechBubble.prototype = Object.create(Phaser.Sprite.prototype);
219 THEGAME.SpeechBubble.prototype.constructor = THEGAME.SpeechBubble;

221 THEGAME.SpeechBubble.wrapBitmapText = function (bitmapText, maxWidth) {
222     var words = bitmapText.text.split(' '), output = "", test = "";
223     
224     for (var w = 0, len = words.length; w < len; w++) {
225         test += words[w] + " ";
226         bitmapText.text = test;
227         bitmapText.updateText();
228         if (bitmapText.textWidth > maxWidth) {
229             output += "\n" + words[w] + " ";
230         }
231         else {
232             output += words[w] + " ";
233         }
234         test = output;
235     }
236     
237     output = output.replace(/(\s)$/gm, ""); // remove trailing spaces
238     bitmapText.text = output;
239     bitmapText.updateText();
240 }
